# [两数之和](https://leetcode-cn.com/problems/two-sum/)

最搓的

```js
/**
 * @param {number[]} nums
 * @param {number} target
 * @return {number[]}
 */
var twoSum = function(nums, target) {
    if(nums.length == 0) return [];
    for(var i =0;i<nums.length-1;i++){
        for(var j =i+1;j<nums.length;j++){
            if(nums[i]+nums[j] == target){
                return [i,j]
            }
        }
    }
    return [];
};
```
