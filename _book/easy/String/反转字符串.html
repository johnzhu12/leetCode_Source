反转字符串

反转字符串

思路

  1. js有个API，直接翻转
var reverseString = function(s) {
  return s.reverse()
};
  1. 倒着遍历，push进数组，然后把之前的删掉
var reverseString = function (s) {
    let oldL = s.length
    for (let len = s.length, i = len - 1; i >= 0; i--) {
        let item = s[i]
        s.push(item)
    }
    s.splice(0, oldL)
};
  1. 镜像循环长度length/2,交换位置
var reverseString = function (s) {
    let len = Math.floor(s.length / 2)
    for (let i = 0; i < len; i++) {
        let item = s[i]
        //交换位置
        let temp = s[i];
        s[i] = s[s.length - i - 1];
        s[s.length - i - 1] = temp;
    }
};
